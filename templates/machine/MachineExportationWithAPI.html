Getting CSV (python)


import requests

API_KEY="{{request.user.APIKey}}"
URL='http://ai.ixioo.com/api/machine/{{Machine_ID}}/InputLines?format=csv'

with requests.session() as client:
    response = client.get( URL, headers={"Authorization": f"Token {API_KEY}"} )
    print( "response.status_code: ", response.status_code )
    print( "response.text:" )
    print( response.text )

Getting XLS (python)


import requests

API_KEY="{{request.user.APIKey}}"
URL='http://ai.ixioo.com/api/machine/{{Machine_ID}}/InputLines?format=xls'
outfile = "/tmp/1.xls"

with requests.session() as client:
    response = client.get( URL, headers={"Authorization": f"Token {API_KEY}"} )
    print( "response.status_code: ", response.status_code )
    print( "response.content length:", len(response.content), "bytes" )

    with open( outfile, 'wb' ) as f:
        f.write( response.content )
